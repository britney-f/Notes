def main(nums, flags, n, w):
    nums.sort()
    count = 0

    for i in range(n-1, -1, -1):
        if flags[i]:
            flags[i] = False
            m = w - nums[i]
            for j in range(i, -1, -1):
                if flags[j] and nums[j] <= m:
                        flags[j] = False
                        break
            count += 1

    return count

s = raw_input()
[n, w] = [int(i) for i in s.split(' ')]
nums = []
flags = []
for i in range(n):
    nums.append(int(raw_input()))
    flags.append(True)

print main(nums, flags, n, w)

#include<iostream>
#include<algorithm>
#include"stdio.h"
#include<cmath>
#include <unordered_map>
#include<string>

using namespace std;

int a[10005];
int nums[10005];

int main()
{
    int n, m, k;
    cin >> n >> m >> k;
    for(int i = 0; i<k; i++)
        cin >> a[i];

    int count = 0;
    int index=0;

    unordered_map<int, int> kv;

    while(kv.size() != m && index < k)
    {
        if(kv.find(a[index]) == kv.end())
        {
            count++;
            kv[a[index]]++;
        }

        index++;
    }

    for(int i=0; i<m; i++)
        nums[a[i+index]]++;

    while(index < k)
    {
        if(kv.count(a[index]) > 0)
        {
            ;
        }
        else
        {
            count++;
            for(auto e: kv)
            {
                if(nums[e.first] == 0)
                {
                    kv.erase(e.first);
                    break;
                }
            }
            kv[a[index]]++;


        }
        nums[a[index]]--;
        if(index + m < k)
            nums[a[index + m]]++;
        index++;
    }
    cout << count << endl;

}
