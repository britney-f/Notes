const int MAX_N = 1001;
const int MAX_M = 1001;

int dp[MAX_N][MAX_M];
char str1[MAX_N];
char str2[MAX_M];

int main()
{

    int n, m;

    cin >> n;
    for(int i = 0; i < n; i++)
        cin >> str1[i];

    cin >> m;
    for(int i = 0; i < m; i++)
        cin >> str2[i];

    memset(dp, 0, sizeof(dp));

    int res = 0;

    for(int i = 0; i < n; i++)
        for(int j = 0; j < m; j++)
    {
        if(str1[i] == str2[j])
        {
            dp[i+1][j+1] = dp[i][j] + 1;
            res = max(res, dp[i+1][j+1]);
        }

    }

    cout << res << endl;


    return 0;
}

#include<stdio.h>

#define inf 1e7
int a[105],b[105];
int dp[10005][105];

/*

int main(){
    int n;
    while(scanf("%d",&n)!=EOF){
        int sum = 0;
        for(int i=0;i<n;i++){
            scanf("%d",&a[i]);
            sum+=a[i];
        }
        for(int i=0;i<n;i++)scanf("%d",&b[i]);

        for(int i=0;i<=10000;i++){
            for(int j=0;j<=n;j++){
                dp[i][j]=-1;
            }
            
        }
        dp[0][0] = 0;
        int min_k = 1000;
        for(int i=0;i<n;i++){

            for(int j=sum+105;j>=b[i];j--){
                for(int k=n;k>0;k--)
                if(dp[j-b[i]][k-1]!=-1){
                    dp[j][k] = max(dp[j][k], dp[j-b[i]][k-1]+a[i]);
                    if (j>=sum)
                        min_k = min(min_k, k);
                }
            }
        }

        int min_c = 0;
        for(int j=sum;j<=sum+105;j++){
            if(dp[j][min_k]!=-1){
                min_c = max(min_c, dp[j][min_k]);
            }
        }
        printf("%d %d\n", min_k, sum-min_c);
    }

    return 0;
}
