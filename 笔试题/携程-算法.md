```c++
#include <bits/stdc++.h>
using namespace  std;

// aabbcddc
int main()
{
	string s;
	cin >> s;
	int last = -1;
	int last_idx_arr[300];
	vector<int> ans;
	for(int i = 0; i < s.size(); i++) {
		last_idx_arr[s[i]] = i;
	}
	int pos = 0, cnt = 0;
	while(pos < s.size()) {
		cnt = 1;
		last = max(last, last_idx_arr[s[pos]]);
		while(pos < last) {
                        ++pos, ++cnt;
			last = max(last, last_idx_arr[s[pos]]);
		}
		ans.push_back(cnt);
                ++pos;
		cnt = 1;
	}
	cout << ans[0];
	for(int i = 1; i < ans.size(); ++i) {
		cout << "," << ans[i];
	}
	cout << endl;
	system("pause");
	return 0;
}
```

```c++
#include <bits/stdc++.h>
using namespace  std;


int main()
{
        vector<double> p_score, n_score;
        double x;
        int n, y;
        cin >> n;
        for (int i = 0; i < n; ++i) {
                cin >> y >> x;
                if(y) p_score.push_back(x);
                else n_score.push_back(x);
        }
        sort(p_score.begin(), p_score.end());
        sort(n_score.begin(), n_score.end(), greater<double>());
        int cnt = 0, i, j;
        for(i = 0; i < p_score.size(); i++) {
                for(j = 0; j < n_score.size(); j++) {
                        if(p_score[i] > n_score[j]) {
                                break;
                        }
                }
                cnt += n_score.size() - j;
        }
        printf("%.2lf\n", cnt * 1.0 / p_score.size() / n_score.size());
	//system("pause");
	return 0;
}
```
